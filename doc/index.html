Standards-Compliant

// Parse and serialize Turtle
final rdf = RdfCore.withStandardFormats();
final doc = '''
  @prefix ex:  .
  @prefix foaf:  .

  ex:alice foaf:knows ex:bob;
    foaf:name "Alice" .
  ex:bob foaf:name "Bob" .
''';
final parsed = rdf.parse(doc);
final turtle = rdf.serialize(parsed);

// Outputs nearly the same as doc, but we did not provide customPrefixes for the serializer so this time we get the full IRIs for example.com.
// Note that well known prefixes like foaf: are used automatically.
// If we wanted exactly the same, we would have to call it like this:
// final turtle = rdf.serialize(parsed, customPrefixes: {'ex': 'http://example.org/'} );
print(turtle);
