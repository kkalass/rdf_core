import 'package:rdf_core/rdf_core.dart';

final graph = RdfGraph();

// Parse and serialize Turtle
final rdf = RdfCore.withStandardFormats();
final doc = '''
  @prefix ex:  .
  ex:s ex:p "o" .
''';
final parsed = rdf.parse(doc, contentType: 'text/turtle');
final turtle = rdf.serialize(parsed, contentType: 'text/turtle');
print(turtle); // Outputs the same as doc
